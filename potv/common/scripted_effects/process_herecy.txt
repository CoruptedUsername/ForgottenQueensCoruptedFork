add_herecy_preacher = {
					random_pop = {
						limit = { 
						NOT = {has_pop_flag = heretic_preacher } 
						pop_is_heretic = yes
						}
						set_pop_flag = heretic_preacher
					}
}
remove_herecy_preacher = {
		  random_pop = {
				limit = { has_pop_flag = heretic_preacher }
				remove_pop_flag = heretic_preacher
			}
}

			limit = {
			AND = {
