# Bad Opinion of religions
trigger_opinion_of_religions = {
	trigger = {
		is_country_type = default
		NOT = { has_ethic = ethic_gestalt_consciousness }
		FROM = { NOT = { has_ethic = ethic_gestalt_consciousness } }
	}
	opinion = {
		base = 0
		
		modifier = {
			add = -20
			has_modifier = "em_atheism"
			FROM = { 
				OR = {
					has_modifier = "em_animism"
					has_modifier = "em_monotheism"
					has_modifier = "em_god_emperor"
					has_modifier = "em_polytheism"
					has_modifier = "em_pantheism"
				}
			}
		}
		modifier = {
			add = -40
			has_modifier = "em_materialism"
			FROM = { 
				OR = {
					has_modifier = "em_animism"
					has_modifier = "em_monotheism"
					has_modifier = "em_god_emperor"
					has_modifier = "em_polytheism"
					has_modifier = "em_pantheism"
				}
			}
		}
	}
}

# Religion spread
trigger_opinion_of_religion_spread = {
	trigger = {
		is_country_type = default
		NOT = { has_modifier = "em_secularism" }
		NOT = { has_ethic = ethic_gestalt_consciousness }
		FROM = { NOT = { has_ethic = ethic_gestalt_consciousness } }
	}
	opinion = {
		base = 0
		
		modifier = {
			add = 100
			FROM = { has_policy_flag = religion_spread }
			OR = {
				AND = {
					has_modifier = "em_animism"
					FROM = { has_modifier = "em_animism" }
				}
				AND = {
					has_modifier = "em_monotheism"
					FROM = { has_modifier = "em_monotheism" }
				}
				AND = {
					has_modifier = "em_polytheism"
					FROM = { has_modifier = "em_polytheism" }
				}
				AND = {
					has_modifier = "em_pantheism"
					FROM = { has_modifier = "em_pantheism" }
				}
				AND = {
					has_modifier = "em_atheism"
					FROM = { has_modifier = "em_atheism" }
				}
				AND = {
					has_modifier = "em_materialism"
					FROM = { has_modifier = "em_materialism" }
				}
			}
		}
		modifier = {
			add = -100
			FROM = { has_policy_flag = religion_spread }
			OR = {
				AND = {
					NOT = { has_modifier = "em_animism" }
					FROM = { has_modifier = "em_animism" }
				}
				AND = {
					NOT = { has_modifier = "em_monotheism" }
					FROM = { has_modifier = "em_monotheism" }
				}
				AND = {
					NOT = { has_modifier = "em_polytheism" }
					FROM = { has_modifier = "em_polytheism" }
				}
				AND = {
					NOT = { has_modifier = "em_pantheism" }
					FROM = { has_modifier = "em_pantheism" }
				}
				AND = {
					NOT = { has_modifier = "em_atheism" }
					FROM = { has_modifier = "em_atheism" }
				}
				AND = {
					NOT = { has_modifier = "em_materialism" }
					FROM = { has_modifier = "em_materialism" }
				}
				AND = {
					NOT = { has_modifier = "em_god_emperor" }
					FROM = { has_modifier = "em_god_emperor" }
				}
			}
		}
	}
}

# Religion conversion
trigger_opinion_of_subject_religion_conversion = {
	trigger = {
		is_country_type = default
		NOT = { has_ethic = ethic_gestalt_consciousness }
		FROM = { NOT = { has_ethic = ethic_gestalt_consciousness } }
		exists = overlord
		overlord = { is_same_value = FROM }
		has_country_flag = religion_converted
	}
	opinion = {
		base = -100
	}
}
