namespace = potv_religions

### Hidden Game Start Event
event = {
	id = potv_religions.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		

		every_country = {
		limit = { is_country_type = default }
		add_modifier = { modifier = religious_country days = -1}
		}

	}
}



planet_event = {
	id = potv_religions.2
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = { ## no religion ethic, set none
		limit = {
		owner = {
		NOR = {
				has_ethic = ethic_atheism 
				has_ethic = ethic_agnosticism 
				has_ethic = ethic_deism 
				has_ethic = ethic_polytheism 
				has_ethic = ethic_animatism 
				has_ethic = ethic_animism 
				}
				}
		}
		owner = { country_add_ethic = ethic_no_religion }
		
		}
		
		if = { ## no religion ethic, set none
			limit = {
			owner = {
			NOT = { has_ethic = ethic_gestalt_consciousness }
			}
			}
			every_owned_pop = {
			pop_add_country_religion = yes
			}
		}
	}
  }
  
  
# A pop has been added, set religion
# Root = pop
# From = planet
pop_event = {
	id = potv_religions.3
	hide_window = yes
	is_triggered_only = yes

	immediate = {	
	if = {limit = {pop_has_empty_religion = yes} pop_add_country_religion = yes }
		
		 }
}
